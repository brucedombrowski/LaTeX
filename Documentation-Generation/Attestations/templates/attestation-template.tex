%% attestation-template.tex - Software/Tool Attestation Template
%% LaTeX Template for documenting software versions, checksums, and verification
%%
%% This file is meant to be \input after defining required variables.
%% Do not compile this file directly - use a wrapper like software_attestation.tex
%%
%% Required variables (define before \input):
%%   \attDocTitle      - Document title (e.g., "Software Attestation")
%%   \attDocId         - Document ID (e.g., "ATT-2026-001")
%%   \attDate          - Date of attestation
%%   \attAuthor        - Author/attestor name
%%   \attPurpose       - Purpose statement
%%   \attContent       - Main content (software table, checksums, etc.)
%%
%% Optional variables:
%%   \attToolkitVersion - Toolkit version used to generate
%%   \attToolkitCommit  - Git commit hash

\usepackage[margin=1in]{geometry}
\usepackage{fancyhdr}
\setlength{\headheight}{14pt}
\usepackage{lastpage}
\usepackage{booktabs}
\usepackage{tabularx}
\usepackage{longtable}
\usepackage{enumitem}
\usepackage{xcolor}
\usepackage{hyperref}

%% =============================================================================
%% COLORS
%% =============================================================================
\definecolor{headerblue}{RGB}{0,51,102}
\definecolor{attgreen}{RGB}{0,128,0}
\definecolor{attgray}{RGB}{128,128,128}

%% =============================================================================
%% HEADER/FOOTER
%% =============================================================================
\pagestyle{fancy}
\fancyhf{}
\fancyhead[L]{\textbf{\attDocTitle}}
\fancyhead[R]{\attDocId}
\fancyfoot[L]{\small\textcolor{attgray}{Generated by LaTeX Toolkit}}
\fancyfoot[C]{\thepage\ of \pageref{LastPage}}
\fancyfoot[R]{\small\textcolor{attgray}{\attDate}}
\renewcommand{\headrulewidth}{0.4pt}
\renewcommand{\footrulewidth}{0.4pt}

%% =============================================================================
%% HYPERREF SETUP
%% =============================================================================
\hypersetup{
    colorlinks=true,
    linkcolor=headerblue,
    urlcolor=headerblue,
    pdftitle={\attDocTitle},
    pdfauthor={\attAuthor}
}

%% =============================================================================
%% DOCUMENT
%% =============================================================================
\begin{document}

%% Title Block
\begin{center}
    {\Large\textbf{\attDocTitle}}\\[0.5em]
    {\normalsize Document ID: \attDocId}\\[0.3em]
    {\normalsize Date: \attDate}\\[0.3em]
    {\normalsize Author: \attAuthor}
\end{center}

\vspace{1em}
\hrule
\vspace{1em}

%% Purpose Section
\section*{Purpose}
\attPurpose

%% Main Content
\attContent

%% Verification Section
\section*{Verification}
This attestation documents the software versions and checksums at the time of generation.
To verify:
\begin{enumerate}
    \item Compare the documented checksums against the actual files
    \item Verify the digital signature on this PDF (if signed)
    \item Confirm the download URLs match official release sources
\end{enumerate}

%% Attestation Block
\section*{Attestation}
I hereby attest that the information documented above is accurate and complete
to the best of my knowledge as of the date specified.

\vspace{2em}

\begin{tabular}{@{}ll@{}}
    \textbf{Attestor:} & \attAuthor \\[0.5em]
    \textbf{Date:} & \attDate \\[0.5em]
    \textbf{Document ID:} & \attDocId \\
\end{tabular}
